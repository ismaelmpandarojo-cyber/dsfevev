<script>

    let currentStep = 0;

    const heartbeat = document.getElementById('heartbeat-audio');

    const scareAudio = document.getElementById('scare-audio');

    const divineAudio = document.getElementById('divine-audio');

    const heavenAudio = document.getElementById('heaven-audio');

 

    const gameSteps = [

        {

            text: "¿Cómo se llamaba la parcela cedida en 1941 para el edificio definitivo?",

            options: [

                { text: "Parcela del Carmen", correct: true },

                { text: "Terreno de Coia", correct: false },

                { text: "Finca de Bellavista", correct: false }

            ]

        },

        {

            text: "¿Qué industria hizo rico a José Policarpo Sanz en Nueva York?",

            options: [

                { text: "Industria del acero", correct: false },

                { text: "Industria del caucho", correct: true },

                { text: "Comercio de café", correct: false }

            ]

        },

        {

            text: "La torre norte imita a un templo famoso en París. ¿Cuál?",

            options: [

                { text: "Notre Dame", correct: false },

                { text: "Saint-Sulpice", correct: true },

                { text: "Sacre-Coeur", correct: false }

            ]

        },

        {

            text: "¿Quién fue el arquitecto que diseñó este palacio de la enseñanza?",

            options: [

                { text: "Antonio Palacios", correct: false },

                { text: "Manuel Gómez Román", correct: true },

                { text: "Jenaro de la Fuente", correct: false }

            ]

        },

        {

            text: "¿En honor a quién puso Policarpo Sanz el nombre de 'Santa Irene'?",

            options: [

                { text: "A su madre", correct: false },

                { text: "A su esposa", correct: true },

                { text: "A su hija fallecida", correct: false }

            ]

        },

        {

            text: "¿Desde qué año se celebran oficialmente las campanadas de Vigo en este reloj?",

            options: [

                { text: "Desde 1946", correct: false },

                { text: "Desde 1999", correct: true },

                { text: "Desde 1927", correct: false }

            ]

        },

        {

            text: "¿Cuál es el lema que preside el espíritu del edificio?",

            options: [

                { text: "Cultura y Libertad", correct: false },

                { text: "Trabajo y Constancia", correct: false },

                { text: "Instrucción y Caridad", correct: true }

            ]

        }

    ];

 

    function startGame() {

        // Unlock audio context

        const context = new (window.AudioContext || window.webkitAudioContext)();

        if (context.state === 'suspended') {

            context.resume();

        }

 

        document.getElementById('screen-start').classList.add('hidden');

        document.getElementById('screen-play').classList.remove('hidden');

        heartbeat.volume = 0.5;

        heartbeat.play().catch(e => console.log("Audio play prevented"));

        showStep();

    }

 

    function showStep() {

        const step = gameSteps[currentStep];

        const progress = (currentStep / gameSteps.length) * 100;

        document.getElementById('progress-fill').style.width = progress + "%";

       

        document.getElementById('story-text').innerText = step.text;

        const container = document.getElementById('options-container');

        container.innerHTML = '';

        step.options.forEach(opt => {

            const btn = document.createElement('button');

            btn.className = "w-full p-4 border border-zinc-800 bg-zinc-900/60 hover:bg-red-950/40 hover:border-red-700 transition-all text-center uppercase tracking-widest text-sm md:text-base font-bold";

            btn.innerText = opt.text;

            btn.onclick = () => checkAnswer(opt.correct);

            container.appendChild(btn);

        });

       

        heartbeat.playbackRate = 1.0 + (currentStep * 0.15);

    }

 

    function checkAnswer(isCorrect) {

        if (isCorrect) {

            currentStep++;

            if (currentStep < gameSteps.length) {

                showStep();

            } else {

                document.getElementById('progress-fill').style.width = "100%";

                winGame();

            }

        } else {

            failGame();

        }

    }

 

    function failGame() {

        heartbeat.pause();

        scareAudio.currentTime = 0;

        scareAudio.volume = 1.0;

        scareAudio.play();

        const overlay = document.getElementById('jumpscare');

        overlay.style.display = 'flex';

        document.body.classList.add('flash');

        setTimeout(() => {

            overlay.style.display = 'none';

            document.body.classList.remove('flash');

            heartbeat.play();

        }, 1200);

    }

 

    function winGame() {

        heartbeat.pause();

        divineAudio.volume = 0.7;

        divineAudio.play();

        const hallway = document.getElementById('divine-hallway');

        hallway.style.display = 'flex';

        setTimeout(() => {

            document.getElementById('victory-content').classList.remove('hidden');

        }, 2000);

    }

 

    function ascendToHeaven() {

        const flashDiv = document.createElement('div');

        flashDiv.className = "fixed inset-0 z-[3000] soft-white-transition";

        document.body.appendChild(flashDiv);

 

        setTimeout(() => {

            document.getElementById('divine-hallway').style.display = 'none';

            const heaven = document.getElementById('heaven-scene');

            heaven.style.display = 'flex';

            divineAudio.pause();

           

            // Majestic celestial choral sound

            heavenAudio.volume = 1.0;

            heavenAudio.play().catch(e => console.log("Audio play error"));

           

            flashDiv.style.opacity = '0';

            setTimeout(() => flashDiv.remove(), 2000);

 

            setTimeout(() => {

                document.getElementById('figure').style.opacity = '1';

                setTimeout(() => {

                    document.getElementById('heaven-text').style.opacity = '1';

                }, 1500);

            }, 1000);

        }, 1500);

    }
